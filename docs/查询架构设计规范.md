# 查询架构设计规范

## 1. 概述

本文档定义了查询层的架构设计规范。查询层是独立于UI实现的核心数据查询系统,位于src/query目录下,为整个应用提供统一的数据访问接口。查询层负责实现数据查询逻辑、与数据库实现层交互、管理缓存数据以及实现复杂业务逻辑。

### 1.1 架构目标

- **数据访问统一化**:所有数据访问通过查询层进行,避免应用层直接与数据库实现层交互
- **性能优化**:通过缓存层实现数据的快速访问,减少数据库访问次数
- **职责分离**:查询层职责明确,边界清晰,便于维护和扩展
- **业务逻辑封装**:将复杂的数据查询和业务逻辑封装在查询层
- **独立性**:查询层独立于UI实现,可被任何模块复用

### 1.2 架构层次

```
┌─────────────────────────────────────┐
│         应用层 (app)                 │
│  - UI渲染                            │
│  - 用户交互                          │
│  - 业务流程控制                      │
└──────────────┬──────────────────────┘
               │ 调用查询层
┌──────────────▼──────────────────────┐
│         查询层 (query)               │
│  - 数据查询                          │
│  - 复杂业务逻辑                      │
│  - 缓存管理                          │
└──────────────┬──────────────────────┘
               │ 访问数据库
┌──────────────▼──────────────────────┐
│      数据库实现层 (database)         │
│  - 数据持久化                        │
│  - 数据库操作                        │
└─────────────────────────────────────┘
```

## 3. 查询层职责

### 3.1 主要职责

- 实现数据查询逻辑
- 与数据库实现层交互
- 管理缓存数据
- 实现复杂业务逻辑
- 提供统一的数据访问接口

### 3.2 能力边界

**可以做的:**
- 调用数据库实现层获取数据
- 管理和更新缓存
- 实现复杂的数据查询和过滤
- 实现数据聚合和转换
- 提供预加载和批量查询功能

**不可以做的:**
- 直接操作DOM
- 处理用户交互事件
- 实现UI渲染逻辑
- 直接访问浏览器存储(IndexedDB除外)

```typescript
/**

 * 定义查询的输入参数

export interface QueryParams {
  /** 页码 */
  page?: number;
  /** 每页大小 */
  pageSize?: number;
  /** 过滤条件 */
  filters?: FilterCondition[];
  /** 排序条件 */
  sort?: SortCondition[];
  /** 搜索关键词 */
  keyword?: string;
  /** 其他自定义参数 */
  [key: string]: any;
}

/**
 * 过滤条件
 */
export interface FilterCondition {
  /** 字段名 */
  field: string;
  /** 操作符 */
  operator: FilterOperator;
  /** 值 */
  value: any;
}

/**
 * 过滤操作符
 */
export enum FilterOperator {
  EQUALS = "eq",
  NOT_EQUALS = "ne",
  GREATE_THAN = "gt",
  GREATER_THAN_OR_EQUAL = "gte",
  LESS_THAN = "lt",
  LESS_THAN_OR_EQUAL = "lte",
  CONTAINS = "contains",
  STARTS_WITH = "startsWith",
  ENDS_WITH = "endsWith",
  IN = "in",
  NOT_IN = "notIn"
}

/**
 * 排序条件
 */
export interface SortCondition {
  /** 字段名 */
  field: string;
  /** 排序方向 */
  direction: "asc" | "desc";
}

/**
 * 查询结果
 * 定义查询的返回结果
 */
export interface QueryResult<T> {
  /** 数据列表 */
  data: T[];
  /** 总数 */
  total: number;
  /** 当前页 */
  page: number;
  /** 每页大小 */
  pageSize: number;
  /** 是否有下一页 */
  hasNext: boolean;
  /** 是否有上一页 */
  hasPrev: boolean;
}

/**
 * 分页参数
 */
export interface PaginationParams {
  page: number;
  pageSize: number;
}

/**
 * 查询选项
 */
export interface QueryOptions {
  /** 是否使用缓存 */
  useCache?: boolean;
  /** 是否预加载 */
  preload?: boolean;
  /** 预加载数量 */
  preloadCount?: number;
```

### 4.2 查询实现

```typescript
// [resource]-query.ts

import { QueryParams, QueryResult, QueryOptions, FilterCondition, SortCondition } from "../types.js";
import { ResourceRepository } from "../../database/implementations/[resource]-repository.impl.js";
import { cacheManager } from "../utils/cache-manager.js";

/**
 * 查询错误类
 */
export class QueryError extends Error {
  constructor(message: string, public readonly cause?: Error) {
    super(message);
    this.name = "QueryError";
  }
}

/**
 * 资源查询类
 */
export class ResourceQuery {
  private repository: ResourceRepository;

  constructor() {
    this.repository = new ResourceRepository();
  }

  /**
   * 获取资源列表
   * @param params 查询参数
   * @param options 查询选项
   * @returns 查询结果
   */
  async getResources(
    params: QueryParams,
    options: QueryOptions = {}
  ): Promise<QueryResult<ResourceType>> {
    try {
      // 检查缓存
      if (options.useCache !== false) {
        const cached = cacheManager.get<ResourceType[]>(this.buildCacheKey(params));
        if (cached) {
          return this.buildQueryResult(cached, params);
        }
      }

      // 从数据库获取数据
      const data = await this.repository.find(params);

      // 更新缓存
      if (options.useCache !== false) {
        cacheManager.set(this.buildCacheKey(params), data);
      }

      // 预加载
      if (options.preload && options.preloadCount) {
        this.preloadNextPage(params, options.preloadCount);
      }

      return this.buildQueryResult(data, params);
    } catch (error) {
      console.error("[Query] Get resources error:", error);
      throw new QueryError("获取资源列表失败", error as Error);
    }
  }

  /**
   * 根据ID获取资源
   * @param id 资源ID
   * @param options 查询选项
   * @returns 资源对象
   */
  async getResourceById(
    id: string,
    options: QueryOptions = {}
  ): Promise<ResourceType | undefined> {
    try {
      // 检查缓存
      if (options.useCache !== false) {
        const cached = cacheManager.get<ResourceType>(`resource:${id}`);
        if (cached) {
          return cached;
        }
      }

      // 从数据库获取数据
      const resource = await this.repository.findById(id);

      // 更新缓存
      if (resource && options.useCache !== false) {
        cacheManager.set(`resource:${id}`, resource);
      }

      return resource;
    } catch (error) {
      console.error("[Query] Get resource by id error:", error);
      throw new QueryError("获取资源失败", error as Error);
    }
  }

  /**
   * 批量获取资源
   * @param ids 资源ID列表
   * @param options 查询选项
   * @returns 资源对象数组
   */
  async getResourcesByIds(
    ids: string[],
    options: QueryOptions = {}
  ): Promise<ResourceType[]> {
    try {
      // 检查缓存
      const cachedMap = new Map<string, ResourceType>();
      const uncachedIds: string[] = [];

      if (options.useCache !== false) {
        for (const id of ids) {
          const cached = cacheManager.get<ResourceType>(`resource:${id}`);
          if (cached) {
            cachedMap.set(id, cached);
          } else {
            uncachedIds.push(id);
          }
        }
      } else {
        uncachedIds.push(...ids);
      }

      // 从数据库获取未缓存的数据
      let resources: ResourceType[] = [];
      if (uncachedIds.length > 0) {
        resources = await this.repository.findByIds(uncachedIds);

        // 更新缓存
        if (options.useCache !== false) {
          for (const resource of resources) {
            cacheManager.set(`resource:${resource.id}`, resource);
          }
        }
      }

      // 合并缓存和数据库数据
      const result: ResourceType[] = [];
      for (const id of ids) {
        const cached = cachedMap.get(id);
        if (cached) {
          result.push(cached);
        } else {
          const resource = resources.find(r => r.id === id);
          if (resource) {
            result.push(resource);
          }
        }
      }

      return result;
    } catch (error) {
      console.error("[Query] Get resources by ids error:", error);
      throw new QueryError("批量获取资源失败", error as Error);
    }
  }

  /**
   * 构建资源索引
   * @param options 索引构建选项
   */
  async buildResourceIndex(options?: IndexOptions): Promise<void> {
    try {
      // 获取所有资源
      const resources = await this.repository.findAll();

      // 构建索引
      const indices = resources.map(resource => ({
        id: resource.id,
        title: resource.title,
        tags: resource.tags,
        // 其他索引字段
      }));

      // 更新缓存
      cacheManager.set("resource:index", indices);
    } catch (error) {
      console.error("[Query] Build resource index error:", error);
      throw new QueryError("构建资源索引失败", error as Error);
    }
  }

  /**
   * 搜索资源
   * @param keyword 搜索关键词
   * @param params 查询参数
   * @returns 查询结果
   */
  async searchResources(
    keyword: string,
    params: QueryParams = {}
  ): Promise<QueryResult<ResourceType>> {
    try {
      // 获取索引
      const indices = cacheManager.get<IndexItem[]>("resource:index");
      if (!indices) {
        throw new QueryError("索引未构建,请先调用buildResourceIndex");
      }

      // 执行搜索
      const matchedIds = this.searchIndex(keyword, indices);

      // 获取匹配的资源
      const resources = await this.getResourcesByIds(matchedIds, { useCache: true });

      return this.buildQueryResult(resources, params);
    } catch (error) {
      console.error("[Query] Search resources error:", error);
      throw new QueryError("搜索资源失败", error as Error);
    }
  }

  /**
   * 构建缓存键
   * @param params 查询参数
   * @returns 缓存键
   */
  private buildCacheKey(params: QueryParams): string {
    const keyParts = [
      "resource",
      params.page || 0,
      params.pageSize || 10,
      JSON.stringify(params.filters || []),
      JSON.stringify(params.sort || []),
      params.keyword || ""
    ];
    return keyParts.join(":");
  }

  /**
   * 构建查询结果
   * @param data 数据列表
   * @param params 查询参数
   * @returns 查询结果
   */
  private buildQueryResult(
    data: ResourceType[],
    params: QueryParams
  ): QueryResult<ResourceType> {
    const page = params.page || 0;
    const pageSize = params.pageSize || 10;
    const total = data.length;
    const totalPages = Math.ceil(total / pageSize);

    return {
      data,
      total,
      page,
      pageSize,
      hasNext: page < totalPages - 1,
      hasPrev: page > 0
    };
  }

  /**
   * 预加载下一页
   * @param params 当前查询参数
   * @param count 预加载数量
   */
  private async preloadNextPage(params: QueryParams, count: number): Promise<void> {
    const nextPageParams = {
      ...params,
      page: (params.page || 0) + 1,
      pageSize: count
    };

    try {
      await this.getResources(nextPageParams, { useCache: true });
    } catch (error) {
      console.error("[Query] Preload next page error:", error);
      // 预加载失败不影响主流程
    }
  }

  /**
   * 搜索索引
   * @param keyword 搜索关键词
   * @param indices 索引列表
   * @returns 匹配的ID列表
   */
  private searchIndex(keyword: string, indices: IndexItem[]): string[] {
    const lowerKeyword = keyword.toLowerCase();
    return indices
      .filter(index =>
        index.title.toLowerCase().includes(lowerKeyword) ||
        index.tags.some(tag => tag.toLowerCase().includes(lowerKeyword))
      )
      .map(index => index.id);
  }
}
```

## 5. 缓存管理

### 5.1 缓存管理器

```typescript
// utils/cache-manager.ts

import { ICache, CacheStats } from "../../cache/index.js";

/**
 * 缓存管理器
 * 负责管理查询层的缓存操作
 */
export class CacheManager {
  private caches: Map<string, ICache<string, any>> = new Map();

  /**
   * 注册缓存
   * @param name 缓存名称
   * @param cache 缓存实例
   */
  registerCache(name: string, cache: ICache<string, any>): void {
    this.caches.set(name, cache);
  }

  /**
   * 获取缓存
   * @param name 缓存名称
   * @returns 缓存实例
   */
  getCache(name: string): ICache<string, any> | undefined {
    return this.caches.get(name);
  }

  /**
   * 获取缓存值
   * @param key 缓存键
   * @returns 缓存值
   */
  get<T>(key: string): T | undefined {
    for (const cache of this.caches.values()) {
      const value = cache.get(key);
      if (value !== undefined) {
        return value as T;
      }
    }
    return undefined;
  }

  /**
   * 设置缓存值
   * @param key 缓存键
   * @param value 缓存值
   */
  set(key: string, value: any): void {
    for (const cache of this.caches.values()) {
      cache.set(key, value);
    }
  }

  /**
   * 删除缓存值
   * @param key 缓存键
   */
  delete(key: string): void {
    for (const cache of this.caches.values()) {
      cache.delete(key);
    }
  }

  /**
   * 清空所有缓存
   */
  clear(): void {
    for (const cache of this.caches.values()) {
      cache.clear();
    }
  }

  /**
   * 获取所有缓存统计
   * @returns 缓存统计信息
   */
  getAllStats(): Map<string, CacheStats> {
    const stats = new Map<string, CacheStats>();
    for (const [name, cache] of this.caches.entries()) {
      stats.set(name, cache.getStats());
    }
    return stats;
  }
}

// 导出单例实例
export const cacheManager = new CacheManager();
```

```typescript
// utils/query-builder.ts

import { QueryParams, FilterCondition, SortCondition, FilterOperator } from "../types.js";

/**
 * 查询构建器
 * 用于构建复杂的查询条件
 */
export class QueryBuilder {
  private params: QueryParams = {};

  /**
   * 设置分页参数
   * @param page 页码
   * @param pageSize 每页大小
   * @returns 查询构建器
   */
  paginate(page: number, pageSize: number): this {
    this.params.page = page;
    this.params.pageSize = pageSize;
    return this;
  }

  /**
   * 添加过滤条件
   * @param field 字段名
   * @param operator 操作符
   * @param value 值
   * @returns 查询构建器
   */
  filter(field: string, operator: FilterOperator, value: any): this {
    if (!this.params.filters) {
      this.params.filters = [];
    }
    this.params.filters.push({ field, operator, value });
    return this;
  }

  /**
   * 添加排序条件
   * @param field 字段名
   * @param direction 排序方向
   * @returns 查询构建器
   */
  sort(field: string, direction: "asc" | "desc" = "asc"): this {
    if (!this.params.sort) {
      this.params.sort = [];
    }
    this.params.sort.push({ field, direction });
    return this;
  }

  /**
   * 设置搜索关键词
   * @param keyword 搜索关键词
   * @returns 查询构建器
   */
  search(keyword: string): this {
    this.params.keyword = keyword;
    return this;
  }

  /**
   * 设置自定义参数
   * @param key 参数名
   * @param value 参数值
   * @returns 查询构建器
   */
  setParam(key: string, value: any): this {
    this.params[key] = value;
    return this;
  }

  /**
   * 构建查询参数
   * @returns 查询参数
   */
  build(): QueryParams {
    return { ...this.params };
  }

  /**
   * 重置查询构建器
   * @returns 查询构建器
   */
  reset(): this {
    this.params = {};
    return this;
  }
}
```

### 7.1 查询层错误处理

```typescript
/**
 * 查询错误处理示例
 */
export async function executeQuery<T>(
  params: QueryParams,
  options: QueryOptions = {}
): Promise<QueryResult<T>> {
  try {
    // 查询逻辑
    const result = await queryExecutor.execute<T>(params, options);
    return result;
  } catch (error) {
    console.error("[Query] Error:", error);

    // 判断错误类型
    if (error instanceof QueryError) {
      // 已知的查询错误
      throw error;
    } else if (error instanceof DatabaseError) {
      // 数据库错误
      throw new QueryError("数据库操作失败", error);
    } else {
      // 未知错误
      throw new QueryError("查询失败", error as Error);
    }
  }
}
```

## 8. 性能优化

### 8.1 查询优化

1. **索引构建**:为频繁查询的字段建立索引
2. **批量查询**:减少数据库访问次数
3. **延迟加载**:非必要数据延迟加载
4. **查询合并**:合并相似查询请求

### 8.2 缓存策略

1. **热点数据缓存**:频繁访问的数据必须缓存
2. **批量操作**:支持批量获取和更新
3. **预加载**:实现下一页数据的预加载
4. **缓存淘汰**:使用LRU策略管理缓存大小

```typescript
// debug.ts

/**
 * 查询调试器
 */
export class QueryDebugger {
  private queries: QueryLog[] = [];

  /**
   * 记录查询
   * @param params 查询参数
   * @param result 查询结果
   * @param duration 执行时间
   */
  logQuery(params: QueryParams, result: QueryResult<any>, duration: number): void {
    this.queries.push({
      timestamp: Date.now(),
      params,
      result,
      duration
    });
  }

  /**
   * 获取查询日志
   * @returns 查询日志列表
   */
  getQueries(): QueryLog[] {
    return [...this.queries];
  }

  /**
   * 清空查询日志
   */
  clear(): void {
    this.queries = [];
  }

  /**
   * 生成查询报告
   * @returns 报告文本
   */
  generateReport(): string {
    let report = "查询性能报告\n";
    report += "=".repeat(50) + "\n\n";

    const totalQueries = this.queries.length;
    const totalDuration = this.queries.reduce((sum, q) => sum + q.duration, 0);
    const avgDuration = totalDuration / totalQueries;
    const maxDuration = Math.max(...this.queries.map(q => q.duration));
    const minDuration = Math.min(...this.queries.map(q => q.duration));

    report += `总查询数: ${totalQueries}\n`;
    report += `总耗时: ${totalDuration.toFixed(2)}ms\n`;
    report += `平均耗时: ${avgDuration.toFixed(2)}ms\n`;
    report += `最大耗时: ${maxDuration.toFixed(2)}ms\n`;
    report += `最小耗时: ${minDuration.toFixed(2)}ms\n\n`;

    report += "慢查询(>100ms):\n";
    const slowQueries = this.queries.filter(q => q.duration > 100);
    if (slowQueries.length === 0) {
      report += "无\n";
    } else {
      slowQueries.forEach(q => {
        report += `- ${JSON.stringify(q.params)}: ${q.duration.toFixed(2)}ms\n`;
      });
    }

    return report;
  }
}

/**
 * 查询日志
 */
interface QueryLog {
  timestamp: number;
  params: QueryParams;
  result: QueryResult<any>;
  duration: number;
}
```
